<script>
    document.addEventListener('DOMContentLoaded', function () {
    const form = document.getElementById('myForm-fin');
    const submitBtn = document.getElementById('submitBtn-fin');
    const loader = document.getElementById('loader-fin');
    const dots = document.getElementById('dots-fin');

    let dotsCount = 0;
    let loaderInterval;

    function updateDots() {
        dotsCount = (dotsCount % 3) + 1;
        dots.textContent = ".".repeat(dotsCount);
    }

    form.addEventListener('submit', function (e) {
        e.preventDefault(); // Prevent form submission

        // Show loader
        loader.classList.add('active');

        // Start dot animation
        updateDots();
        loaderInterval = setInterval(updateDots, 500);

        // Simulate form processing (5 seconds)
        setTimeout(function () {
            // Hide loader
            loader.classList.remove('active');
            clearInterval(loaderInterval);

            // Reset form or redirect as needed
            form.reset();
            alert('Форма успешно отправлена!');
        }, 3000);
        });
    });
</script>
